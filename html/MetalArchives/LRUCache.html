def [](key)
  if @keys.include? key
    MetalArchives.config.logger.debug "Cache hit for #{key}"
    @keys.delete key
    @keys << key
  else
    MetalArchives.config.logger.debug "Cache miss for #{key}"
  end

  @cache[key]
end

def []=(key, value)
  @cache[key] = value

  @keys.delete key if @keys.include? key

  @keys << key

  pop if @keys.size > @size
end
